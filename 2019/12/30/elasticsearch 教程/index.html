测试
kibana Dev Tools

index

创建索引:

PUT http://127.0.0.1:9200/test_index

删除索引

DELETE http://127.0.0.1:9200/test_index

document

创建文档：

PUT http://127.0.0.1:9200/test_index/item/1
{
"name":"jiaopan",
"age":25
}

查询文档：

GET http://127.0.0.1:9200/test_index/item/1

更新文档

POST http://127.0.0.1:9200/test_index/item/1/_update
{
    "doc":{
        "name":"jp"
    }
}

删除文档

DELETE http://127.0.0.1:9200/test_index/item/1

搜索

1.query string search

GET http://127.0.0.1:9200/test_index/item/_search
GET http://127.0.0.1:9200/test_index/item/_search?q=name:jp&sort=age:desc

2.query DSL

#查询所有
GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "match_all":{} 
    }
}
#条件查询
GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "match":{"name":"jp"} 
    },
    "sort":[
        {"age":"desc"}
    ]
}

#分页查询
GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "match_all":{} 
    },
    "from":1,
    "size":2
}
#指定结果的字段
GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "match_all":{} 
    },
    "_source“：["name"]
}

3.query filter

GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "bool":{
            "must":{
                "name":"jp" #name必须包含jp
            },
            "filter":{
                "range":{
                    "age":{"gt":18} #年龄大于18
                }
            }
        } 
    }
}

4.full-text search(全文检索)

GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "match":{
            "name":"jp jiaopan" #匹配jp/jiapan/jp或jiaopan
        } 
    }
}

5.phrase search(短语搜索)

GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "match_phrase":{
            "name":"jp jiaopan" #只匹配jp jiaopan
        } 
    }
}

6.highlight search(高亮搜索结果)

GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "match":{
            "name":"jp" 
        } 
    }，
    “highlight“：{
        "fields":{
            "name":{} #高亮的字段
        }
    }
}

聚合分析

#以"tags"字段分组
GET http://127.0.0.1:9200/test_index/item/_search
{
    "aggs":{
        "group_by_tags":{ //聚合名称(自定义)
            "terms":{
                “field”：“tags” //分组字段
            } 
        } 
    }
}
#先查询再分组
GET http://127.0.0.1:9200/test_index/item/_search
{
    "query":{
        "match":{
            "name":"name" 
        } 
    },
    "aggs":{
        "group_by_tags":{ //聚合名称(自定义)
            "terms":{
                “field”：“tags” //分组字段
            } 
        } 
    }
}
#分组数据再分析
GET http://127.0.0.1:9200/test_index/item/_search
{
    "aggs":{
        "group_by_tags":{ //聚合名称(自定义)
            "terms":{
                “field”：“tags” //分组字段
            },
            "aggs":{
                "avg_price":{"field":"price"}
            } 
        } 
    }
}
GET http://127.0.0.1:9200/test_index/item/_search
{
    "aggs":{
        "group_by_tags":{ //聚合名称(自定义)
            "terms":{
                “field”：“tags”, //分组字段
                "order":{"avg_price":"desc"} //排序
            },
            "aggs":{
                "avg_price":{"field":"price"}
            } 
        } 
    }
}
